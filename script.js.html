let noClickCount = 0;
const yesButton = document.getElementById('yesButton');
const noButton = document.getElementById('noButton');
const voiceMessage = document.getElementById('voiceMessage');
function yesClick() {
voiceMessage.play();
setTimeout(() => {
alert("Yay! Thank you for being my Valentine!");
}, 1000);
}
function noClick() {
noClickCount++;
if (noClickCount < 5) {
yesButton.style.transform = `scale(${1 + noClickCount * 0.2})`;
noButton.innerText = "Can you please click Yes?";
} else {
yesButton.style.transform = 'scale(3)';
}
}
